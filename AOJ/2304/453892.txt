bool EK_Bfs(int start,int end)
{
    queue<int> que;
    bool flag[MAXN];
    memset(flag,false,sizeof(flag));
    memset(p,-1,sizeof(p));
    que.push(start);
    flag[start]=true;
    while(!que.empty())
    {
        int e=que.front();
        if(e==end) return true;
        que.pop();
        for(int i=1;i<=n;++i)
        {
            if(map[e][i]&&!flag[i])
            {
                flag[i]=true;
                p[i]=e;
                que.push(i);
            }
        }
    }
    return false;
}
int EK_MAX(int start,int end)
{
    int u,flow_ans=0; int mn;
    while(EK_Bfs(start,end))
    {
        mn=99999999;
        u=end;
        while(p[u]!=-1)
        {
            mn=min(mn,map[p[u]][u]);
            u=p[u];
        }
        flow_ans+=mn;
        u=end;
        while(p[u]!=-1)
        {
            map[p[u]][u]-=mn;
            map[u][p[u]]+=mn;
            u=p[u];
        }
    }
    return flow_ans;
}
